import os


def human_read_format(size):
    if size < 1024:
        return f"{size}Б"
    elif size < 1024 * 1024:
        return f"{round(size / 1024)}КБ"
    elif size < 1024 * 1024 * 1024:
        return f"{round(size / (1024 * 1024))}МБ"
    else:
        return f"{round(size / (1024 * 1024 * 1024))}ГБ"
